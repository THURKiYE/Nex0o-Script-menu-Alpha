local function createButton(parent, text, position, size, callback)
    local button = Instance.new("TextButton")
    button.Name = text
    button.Size = size
    button.Position = position
    button.BackgroundColor3 = config.secondaryColor
    button.BorderColor3 = config.primaryColor
    button.BorderSizePixel = config.borderWidth
    button.TextColor3 = config.textColor
    button.TextSize = 14
    button.Font = Enum.Font.GothamBold
    button.Text = text
    button.Parent = parent
    
    local function onMouseEnter()
        button.BackgroundColor3 = config.accentColor
    end
    
    local function onMouseLeave()
        button.BackgroundColor3 = config.secondaryColor
    end
    
    button.MouseEnter:Connect(onMouseEnter)
    button.MouseLeave:Connect(onMouseLeave)
    button.MouseButton1Click:Connect(callback)
    
    return button
end

local function createToggleSwitch(parent, text, position, size, callback)
    local switchContainer = Instance.new("Frame")
    switchContainer.Name = text .. "Toggle"
    switchContainer.Size = size
    switchContainer.Position = position
    switchContainer.BackgroundColor3 = config.accentColor
    switchContainer.BorderColor3 = config.primaryColor
    switchContainer.BorderSizePixel = config.borderWidth
    switchContainer.Parent = parent
    
    local label = Instance.new("TextLabel")
    label.Name = "Label"
    label.Size = UDim2.new(0.7, 0, 1, 0)
    label.Position = UDim2.new(0, 10, 0, 0)
    label.BackgroundTransparency = 1
    label.TextColor3 = config.textColor
    label.TextSize = 13
    label.Font = Enum.Font.Gotham
    label.Text = text
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.Parent = switchContainer
    
    local toggle = Instance.new("TextButton")
    toggle.Name = "SwitchButton"
    toggle.Size = UDim2.new(0.25, 0, 0.8, 0)
    toggle.Position = UDim2.new(0.72, 0, 0.1, 0)
    toggle.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
    toggle.BorderColor3 = config.primaryColor
    toggle.BorderSizePixel = 1
    toggle.TextColor3 = config.textColor
    toggle.TextSize = 12
    toggle.Font = Enum.Font.GothamBold
    toggle.Text = "OFF"
    toggle.Parent = switchContainer
    
    local isEnabled = false
    
    toggle.MouseButton1Click:Connect(function()
        isEnabled = not isEnabled
        toggle.Text = isEnabled and "ON" or "OFF"
        toggle.BackgroundColor3 = isEnabled and config.primaryColor or Color3.fromRGB(100, 100, 100)
        toggle.TextColor3 = isEnabled and config.secondaryColor or config.textColor
        callback(isEnabled)
    end)
    
    return switchContainer, toggle
end

        for _, tab in pairs(tabContents) do
            tab.Visible = false
        end
        content.Visible = true
    end)
    
    table.insert(tabContents, content)
    return tabButton, content
end
