-- Fly System
local function startFlying()
    local character = player.Character
    if not character then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not rootPart then return end
    
    -- Create BodyVelocity
    flyBodyVelocity = Instance.new("BodyVelocity")
    flyBodyVelocity.Velocity = Vector3.new(0, 0, 0)
    flyBodyVelocity.MaxForce = Vector3.new(10000, 10000, 10000)
    flyBodyVelocity.P = 10000
    flyBodyVelocity.Parent = rootPart
    
    -- Create BodyGyro
    flyBodyGyro = Instance.new("BodyGyro")
    flyBodyGyro.MaxTorque = Vector3.new(10000, 10000, 10000)
    flyBodyGyro.P = 10000
    flyBodyGyro.D = 100
    flyBodyGyro.CFrame = rootPart.CFrame
    flyBodyGyro.Parent = rootPart
    
    humanoid.PlatformStand = true
    flying = true
    
    -- Flight loop
    flyConnection = RunService.Heartbeat:Connect(function()
        if not flying or not flyBodyVelocity or not flyBodyGyro or not rootPart then
            if flyConnection then flyConnection:Disconnect() end
            return
        end
        
        local direction = Vector3.new()
        
        -- Movement controls
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then
            direction = direction + (rootPart.CFrame.LookVector * flySpeed)
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then
            direction = direction - (rootPart.CFrame.LookVector * flySpeed)
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then
            direction = direction - (rootPart.CFrame.RightVector * flySpeed)
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then
            direction = direction + (rootPart.CFrame.RightVector * flySpeed)
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
            direction = direction + Vector3.new(0, flySpeed, 0)
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
            direction = direction - Vector3.new(0, flySpeed, 0)
        end
        
        flyBodyVelocity.Velocity = direction
        flyBodyGyro.CFrame = CFrame.new(rootPart.Position, rootPart.Position + rootPart.CFrame.LookVector)
    end)
    
    FlyButton.Parent.Parent:FindFirstChild("Text").Text = "FLY [ON]"
    notify("Flight enabled")
end

local function stopFlying()
    flying = false
    
    if flyConnection then
        flyConnection:Disconnect()
        flyConnection = nil
    end
    
    if flyBodyVelocity then
        flyBodyVelocity:Destroy()
        flyBodyVelocity = nil
    end
    
    if flyBodyGyro then
        flyBodyGyro:Destroy()
        flyBodyGyro = nil
    end
    
    local character = player.Character
    if character then
        local humanoid = character:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.PlatformStand = false
        end
    end
    
    FlyButton.Parent.Parent:FindFirstChild("Text").Text = "FLY [OFF]"
    notify("Flight disabled")
end

-- Godmode System
local function enableGodmode()
    godmode = true
    
    local function protectCharacter(char)
        if not char then return end
        
        local humanoid = char:WaitForChild("Humanoid", 5)
        if humanoid then
            -- Store original health
            humanoid.MaxHealth = math.huge
            humanoid.Health = math.huge
            
            -- Auto-heal loop
            healConnection = RunService.Heartbeat:Connect(function()
                if not godmode then
                    healConnection:Disconnect()
                    return
                end
                if humanoid.Health < humanoid.MaxHealth then
                    humanoid.Health = humanoid.MaxHealth
                end
            end)
            
            -- Make invincible
            for _, part in pairs(char:GetChildren()) do
                if part:IsA("BasePart") then
                    part.CanCollide = false
                    part.Material = Enum.Material.Neon
                    part.Color = Color3.fromRGB(255, 255, 0)
                end
            end
        end
    end
    
    -- Protect current character
    if player.Character then
        protectCharacter(player.Character)
    end
    
    -- Protect future characters
    player.CharacterAdded:Connect(protectCharacter)
    
    GodButton.Parent.Parent:FindFirstChild("Text").Text = "GODMODE [ON]"
    notify("Godmode enabled - You are invincible!")
end

local function disableGodmode()
    godmode = false
    
    if healConnection then
        healConnection:Disconnect()
        healConnection = nil
    end
    
    local character = player.Character
    if character then
        local humanoid = character:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.MaxHealth = 100
            humanoid.Health = 100
        end
        
        -- Restore original properties
        for _, part in pairs(character:GetChildren()) do
            if part:IsA("BasePart") then
                part.CanCollide = true
                part.Material = Enum.Material.Plastic
                part.Color = Color3.fromRGB(255, 255, 255)
            end
        end
    end
    
    GodButton.Parent.Parent:FindFirstChild("Text").Text = "GODMODE [OFF]"
    notify("Godmode disabled")
end

-- NoClip System
local function enableNoClip()
    noclip = true
    
    local function clip(character)
        if not character then return end
        
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") and part.CanCollide then
                part.CanCollide = false
            end
        end
    end
    
    -- Initial clip
    if player.Character then
        clip(player.Character)
    end
    
    -- Continuous noclip
    noclipConnection = RunService.Stepped:Connect(function()
        if not noclip then
            noclipConnection:Disconnect()
            return
        end
        if player.Character then
            clip(player.Character)
        end
    end)
    
    NoClipButton.Parent.Parent:FindFirstChild("Text").Text = "NOCLIP [ON]"
    notify("NoClip enabled - Walk through walls")
end

local function disableNoClip()
    noclip = false
    
    if noclipConnection then
        noclipConnection:Disconnect()
        noclipConnection = nil
    end
    
    -- Restore collision
    local character = player.Character
    if character then
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = true
            end
        end
    end
    
    NoClipButton.Parent.Parent:FindFirstChild("Text").Text = "NOCLIP [OFF]"
    notify("NoClip disabled")
end

-- Emote System
local function playEmote(emoteName)
    local character = player.Character
    if not character then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return end
    
    -- Try to load animation
    local animation = Instance.new("Animation")
    animation.AnimationId = "rbxassetid://"
    
    -- Animation IDs for popular emotes
    local emoteIds = {
        jerk = "rbxassetid://5917459365",
        bang = "rbxassetid://5918726674",
        dance = "rbxassetid://5915712537",
        laugh = "rbxassetid://5918805987",
        point = "rbxassetid://5918729294"
    }
    
    if emoteIds[string.lower(emoteName)] then
        animation.AnimationId = emoteIds[string.lower(emoteName)]
    else
        animation.AnimationId = emoteIds.dance -- Default to dance
    end
    
    local animator = humanoid:FindFirstChild("Animator")
    if not animator then
        animator = Instance.new("Animator")
        animator.Parent = humanoid
    end
    
    -- Stop current animations
    for _, track in pairs(animator:GetPlayingAnimationTracks()) do
        track:Stop()
    end
    
    -- Load and play animation
    local animationTrack = animator:LoadAnimation(animation)
    animationTrack:Play()
    
    notify("Playing " .. emoteName .. " emote")
end

-- Teleport System
local function teleportToMouse()
    if mouse.Target then
        local character = player.Character
        if not character then return end
        
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        if not rootPart then return end
        
        -- Get position above target
        local targetPosition = mouse.Hit.Position + Vector3.new(0, 5, 0)
        rootPart.CFrame = CFrame.new(targetPosition)
        
        notify("Teleported to mouse position")
    else
        notify("No valid target found")
    end
end

-- Invisible System
local function toggleInvisible()
    invisible = not invisible
    local character = player.Character
    
    if character then
        if invisible then
            -- Store original transparency
            originalTransparency = {}
            
            -- Make all parts transparent
            for _, part in pairs(character:GetDescendants()) do
                if part:IsA("BasePart") then
                    originalTransparency[part] = part.Transparency
                    part.Transparency = 1
                end
            end
            
            InvisButton.Parent.Parent:FindFirstChild("Text").Text = "INVISIBLE [ON]"
            notify("Invisible mode enabled")
        else
            -- Restore original transparency
            for part, transparency in pairs(originalTransparency) do
                if part:IsA("BasePart") then
                    part.Transparency = transparency
                end
            end
            
            originalTransparency = {}
            InvisButton.Parent.Parent:FindFirstChild("Text").Text = "INVISIBLE [OFF]"
            notify("Invisible mode disabled")
        end
    end
end

-- Insta-Heal System
local function instaHeal()

    if humanoid then
        humanoid.Health = humanoid.MaxHealth
        notify("Health restored to 100%")
    end
end

-- Make GUI draggable
local dragging
local dragInput
local dragStart
local startPos

local function update(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)

-- Keybinds (optional)
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    -- F5 to toggle GUI visibility
    if input.KeyCode == Enum.KeyCode.F5 then
        MainFrame.Visible = not MainFrame.Visible
    end
    
    -- F6 to toggle fly
    if input.KeyCode == Enum.KeyCode.F6 then
        if flying then
            stopFlying()
        else
            startFlying()
        end
    end
end)
