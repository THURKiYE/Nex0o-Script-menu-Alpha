# Nex0o Hub - Advanced Xeno Executor GUI

Hier ist ein vollständiges, modernes Dark-Mode GUI mit grünem Akzent-Design für deinen Xeno Executor:

```lua
-- Nex0o Hub v1.0 - Advanced Executor GUI
-- Dark Theme mit grünem Accent Design

local NexoHub = {}
NexoHub.Version = "1.0"
NexoHub.Config = {
    PrimaryColor = Color3.fromRGB(34, 177, 76),      -- Grün
    SecondaryColor = Color3.fromRGB(20, 20, 20),     -- Dunkel
    AccentColor = Color3.fromRGB(15, 15, 15),        -- Noch dunkler
    TextColor = Color3.fromRGB(255, 255, 255),       -- Weiß
    HoverColor = Color3.fromRGB(45, 200, 90)         -- Heller Grün
}

-- Hauptfenster erstellen
local ScreenGui = Instance.new("ScreenGui")

ScreenGui.ResetOnSpawn = false
ScreenGui.ZIndex = 999
ScreenGui.Parent = game:GetService("CoreGui")

CursorLight.TopSurface = Enum.SurfaceType.Smooth
CursorLight.BottomSurface = Enum.SurfaceType.Smooth
CursorLight.Material = Enum.Material.Neon
CursorLight.Color = NexoHub.Config.PrimaryColor
CursorLight.CanQuery = false
CursorLight.CanTouch = false

local CursorPointLight = Instance.new("PointLight")
CursorPointLight.Color = NexoHub.Config.PrimaryColor
CursorPointLight.Range = 15
CursorPointLight.Brightness = 2
CursorPointLight.Parent = CursorLight

CursorLight.Parent = workspace

-- Haupttitel Bar
local TitleBar = Instance.new("Frame")

TitleBar.Size = UDim2.new(0, 600, 0, 50)
TitleBar.BackgroundColor3 = NexoHub.Config.SecondaryColor

local TitleLabel = Instance.new("TextLabel")

TitleLabel.Size = UDim2.new(0.7, 0, 1, 0)

-- UICorner für abgerundete Ecken
local function CreateCorner(Parent, Radius)
    local Corner = Instance.new("UICorner")
    Corner.CornerRadius = UDim.new(0, Radius)
    Corner.Parent = Parent
    return Corner
end

CreateCorner(TitleBar, 12)
CreateCorner(MainContainer, 12)

-- Fenster verschieben
local UserInputService = game:GetService("UserInputService")
local dragging = false
local dragStart = nil
local windowStart = nil

TitleBar.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        windowStart = TitleBar.Parent.Position
    end
end)

TitleBar.InputEnded:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

UserInputService.InputChanged:Connect(function(input, gameProcessed)
    if dragging and input.UserInputType == Enum.UserInputType.Mouse then
        local delta = input.Position - dragStart
        TitleBar.Parent.Position = windowStart + UDim2.new(0, delta.X, 0, delta.Y)
    end
end)

-- Cursor Light folgen
UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Mouse then
        local camera = workspace.CurrentCamera
        local unitRay = camera:ScreenPointToRay(input.Position.X, input.Position.Y)
        local distance = 50
        CursorLight.Position = unitRay.Origin + unitRay.Direction * distance
    end
end)

-- Tab-System
local Tabs = {}

function NexoHub:CreateTab(TabName, Icon)
    local Tab = {}
    Tab.Name = TabName
    Tab.Icon = Icon or "📦"
    Tab.Elements = {}
    
    -- Tab Button
    local TabButton = Instance.new("TextButton")
    TabButton.Name = TabName
    TabButton.Size = UDim2.new(1, 0, 0, 45)
    TabButton.BackgroundColor3 = NexoHub.Config.SecondaryColor
    TabButton.BorderSizePixel = 1
    TabButton.BorderColor3 = NexoHub.Config.PrimaryColor
    TabButton.TextColor3 = NexoHub.Config.TextColor
    TabButton.Font = Enum.Font.GothamBold
    TabButton.TextSize = 12
    TabButton.Text = Tab.Icon .. " " .. TabName
    TabButton.Parent = TabScroll
    
    CreateCorner(TabButton, 8)
    
    -- Content Frame
    local ContentFrame = Instance.new("Frame")
    ContentFrame.Name = TabName .. "Content"
    ContentFrame.Size = UDim2.new(1, 0, 1, 0)
    ContentFrame.BackgroundTransparency = 1
    ContentFrame.Visible = false
    ContentFrame.Parent = ContentScroll
    
    local UIListLayout = Instance.new("UIListLayout")
    UIListLayout.Padding = UDim.new(0, 10)
    UIListLayout.FillDirection = Enum.FillDirection.Vertical
    UIListLayout.SortOrder = Enum.SortOrder.LayoutOrder
    UIListLayout.Parent = ContentFrame
    
    local UIPadding = Instance.new("UIPadding")
    UIPadding.PaddingLeft = UDim.new(0, 10)
    UIPadding.PaddingRight = UDim.new(0, 10)
    UIPadding.PaddingTop = UDim.new(0, 10)
    UIPadding.PaddingBottom = UDim.new(0, 10)
    UIPadding.Parent = ContentFrame
    
    -- Tab Klick
    TabButton.MouseButton1Click:Connect(function()
        for _, frame in pairs(ContentScroll:GetChildren()) do
            if frame:IsA("Frame") then
                frame.Visible = false
            end
        end
        ContentFrame.Visible = true
        
        -- Button Highlight
        for _, button in pairs(TabScroll:GetChildren()) do
            if button:IsA("TextButton") then
                button.BackgroundColor3 = NexoHub.Config.SecondaryColor
                button.TextColor3 = NexoHub.Config.TextColor
            end
        end
        TabButton.BackgroundColor3 = NexoHub.Config.PrimaryColor
        TabButton.TextColor3 = NexoHub.Config.SecondaryColor
    end)
    
    -- Hover Effekt
    TabButton.MouseEnter:Connect(function()
        if TabButton.BackgroundColor3 ~= NexoHub.Config.PrimaryColor then
            TabButton.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
        end
    end)
    
    TabButton.MouseLeave:Connect(function()
        if TabButton.BackgroundColor3 ~= NexoHub.Config.PrimaryColor then
            TabButton.BackgroundColor3 = NexoHub.Config.SecondaryColor
        end
    end)
    
    Tab.ContentFrame = ContentFrame
    Tab.TabButton = TabButton
    
    -- Funktion zum Erstellen von Elementen
    function Tab:AddLabel(Text)
        local Label = Instance.new("TextLabel")
        Label.Size = UDim2.new(1, 0, 0, 30)
        Label.BackgroundTransparency = 1
        Label.Text = Text
        Label.TextColor3 = NexoHub.Config.PrimaryColor
        Label.Font = Enum.Font.GothamBold
        Label.TextSize = 14
        Label.TextXAlignment = Enum.TextXAlignment.Left
        Label.Parent = self.ContentFrame
        return Label
    end
    
    function Tab:AddButton(Text, Callback)
        local Button = Instance.new("TextButton")
        Button.Size = UDim2.new(1, 0, 0, 40)
        Button.BackgroundColor3 = NexoHub.Config.SecondaryColor
        Button.BorderColor3 = NexoHub.Config.PrimaryColor
        Button.BorderSizePixel = 2
        Button.TextColor3 = NexoHub.Config.PrimaryColor
        Button.Font = Enum.Font.GothamBold
        Button.TextSize = 13
        Button.Text = Text
        Button.Parent = self.ContentFrame
        
        CreateCorner(Button, 8)
        
        Button.MouseButton1Click:Connect(Callback)
        
        Button.MouseEnter:Connect(function()
            Button.BackgroundColor3 = NexoHub.Config.PrimaryColor
            Button.TextColor3 = NexoHub.Config.SecondaryColor
        end)
        
        Button.MouseLeave:Connect(function()
            Button.BackgroundColor3 = NexoHub.Config.SecondaryColor
            Button.TextColor3 = NexoHub.Config.PrimaryColor
        end)
        
        return Button
    end
    
    function Tab:AddToggle(Text, DefaultValue, Callback)

        Container.BorderSizePixel = 2
        Container.Parent = self.ContentFrame
        
        CreateCorner(Container, 8)
        
        local Label = Instance.new("TextLabel")
        Label.Size = UDim2.new(0.7, 0, 1, 0)
        Label.BackgroundTransparency = 1
        Label.Text = Text
        Label.TextColor3 = NexoHub.Config.TextColor
        Label.Font = Enum.Font.GothamBold
        Label.TextSize = 13
        Label.TextXAlignment = Enum.TextXAlignment.Left
        Label.Parent = Container
        
        local ToggleButton = Instance.new("TextButton")
        ToggleButton.Size = UDim2.new(0.25, 0, 0.7, 0)
